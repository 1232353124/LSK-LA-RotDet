import torch
import torch.nn as nn
from mmengine.model import BaseModule
from mmengine.registry import MODELS
from mmengine.logging import MMLogger
from mmengine.runner import load_checkpoint
from functools import partial
from mmrotate.registry import MODELS

def simple_layernorm(dim):
    """替代 LayerNorm，作用于 channels_first 格式的 tensor"""
    return nn.GroupNorm(1, dim)

class DummyBlock(nn.Module):
    """简化后的 UniRepLKNetBlock 占位实现"""
    def __init__(self, dim, kernel_size, drop_path=0., **kwargs):
        super().__init__()
        self.conv = nn.Conv2d(dim, dim, kernel_size, padding=kernel_size // 2, groups=1)
        self.norm = nn.BatchNorm2d(dim)
        self.act = nn.ReLU()

    def forward(self, x):
        return self.act(self.norm(self.conv(x)))

@MODELS.register_module()
class UniRepLKNetBackbone(BaseModule):
    def __init__(self,
                 in_chans=3,
                 depths=(2, 2, 6),
                 dims=(64, 128, 256),
                 kernel_sizes=((3, 3), (5, 5), (7, 7)),
                 init_cfg=None,
                 **kwargs):
        super().__init__(init_cfg=init_cfg)
        self.num_stages = len(depths)
        self.out_channels = dims

        self.downsample_layers = nn.ModuleList()
        self.downsample_layers.append(nn.Sequential(
            nn.Conv2d(in_chans, dims[0], kernel_size=3, stride=2, padding=1),
            simple_layernorm(dims[0])
        ))
        for i in range(1, self.num_stages):
            self.downsample_layers.append(nn.Sequential(
                nn.Conv2d(dims[i - 1], dims[i], kernel_size=3, stride=2, padding=1),
                simple_layernorm(dims[i])
            ))

        self.stages = nn.ModuleList()
        for i in range(self.num_stages):
            stage = nn.Sequential(*[
                DummyBlock(dim=dims[i], kernel_size=kernel_sizes[i][j])
                for j in range(depths[i])
            ])
            self.stages.append(stage)

        # 注册 normX 输出，用于提取多层特征
        for i in range(self.num_stages):
            setattr(self, f'norm{i}', simple_layernorm(dims[i]))

        if self.init_cfg is not None and 'checkpoint' in self.init_cfg:
            self.init_weights()

    def init_weights(self):
        logger = MMLogger.get_current_instance()
        ckpt_path = self.init_cfg.get('checkpoint', None)
        if ckpt_path is None:
            logger.info('[UniRepLKNet] No checkpoint provided, skip loading weights.')
            return
        ckpt = load_checkpoint(self, ckpt_path, logger=logger, map_location='cpu')
        logger.info('[UniRepLKNet] Loaded checkpoint from {}'.format(ckpt_path))

    def forward(self, x):
        outs = []
        for stage_idx in range(self.num_stages):
            x = self.downsample_layers[stage_idx](x)
            x = self.stages[stage_idx](x)
            norm_layer = getattr(self, f'norm{stage_idx}')
            outs.append(norm_layer(x))
        return tuple(outs)
